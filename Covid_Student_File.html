Q5. Find top five countries with highest number of average cases, store them in a new dataframe and Visualize them

In[10]:
# Write your code here
mean_cases = df.groupby('countriesAndTerritories')['cases'].mean()
top_countries = mean_cases.nlargest(5)
for index, (country, avg_cases) in enumerate(top_countries.items(), start=0):
    print(f"{index} {country} {avg_cases:f}")
top_countries_data = df[df['countriesAndTerritories'].isin(top_countries.index)]
sns.barplot(x='countriesAndTerritories', y='cases', data=top_countries_data)
